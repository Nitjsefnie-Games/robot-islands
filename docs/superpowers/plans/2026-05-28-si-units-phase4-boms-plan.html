// SI-units rev-16 §13.4 — per-resource base storage cap on a fresh island.
// Defaults via defaultCapForCategory() below; overrides for sub-calibrated
// resources (helium_3, antimatter_propellant, ai_core, foundation_kit, etc).
export const RESOURCE_BASE_CAP: Readonly<Partial<Record<ResourceId, number>>> = {
  // Sub-calibrated rare resources — overrides below the category-rare default of 1.
  // helium_3: 1 unit = 1 g. Player can hold 1 g per island without storage.
  helium_3: 1,
  // antimatter_propellant: 1 unit = 1 ng. Single ng base cap.
  antimatter_propellant: 1,
  // ai_core: 1 unit = 1 kg single chip. Whole-unit-only — 0 base cap.
  ai_core: 0,
  // foundation_kit / enriched_uranium / refined_uranium: large assemblies.
  foundation_kit: 5,
  enriched_uranium: 5,
  refined_uranium: 5,
};

// rev-16 §13.4 — category defaults when no per-resource override is present.
export function defaultCapForCategory(c: StorageCategory): number {
  switch (c) {
    case 'dry_goods':      return 100;  // small pile / personal inventory
    case 'liquid_gas':     return 100;  // small drum
    case 'temp_sensitive': return 50;   // small refrigerated container
    case 'components':     return 20;   // component count, not mass, is the gameplay constraint
    case 'rare':           return 1;    // precious materials scarce by default
  }
}

11.1 simulateOptimalPath algorithm

BFS over recipe-graph predecessors from the target building. The frontier is "buildings whose placement BOM is satisfied right now"; in each step, place every frontier building, advance the simulator one cycle of recipe-time, expand the frontier with newly-buildable predecessors of the target.

// src/test-helpers/reachability.ts
import { advanceIsland } from '../economy.js';
import { BUILDING_DEFS, type BuildingDefId } from '../building-defs.js';
import { RECIPES } from '../recipes.js';
import type { IslandState, Inventory } from '../economy.js';

export interface ReachabilityOutcome {
  reached: boolean;
  elapsedMs: number;
  placedBuildings: BuildingDefId[];
  blockedBy: BuildingDefId | null;     // the predecessor we ran out of time on
  blockingResource: string | null;
}

export function simulateOptimalPath(
  initial: IslandState,
  deadlineMs: number,
  target: BuildingDefId,
): ReachabilityOutcome {
  // 1. Build dependency closure: BFS over RECIPES predecessors of the
  //    target's placementCost. Stop at resources that the starter
  //    inventory already provides.
  const required = closurePredecessors(target);

  // 2. Pick the minimum set of intermediate buildings that produces
  //    every required resource. Greedy by tier (T0/T1 first).
  const chain = selectBuildingChain(required);

  // 3. Place + advance loop.
  let state = initial;
  let elapsedMs = 0;
  const placed: BuildingDefId[] = [];
  const stepMs = 30 * 1000;  // 30 s simulated steps

  while (elapsedMs < deadlineMs) {
    // Place every chain-building whose placementCost is currently
    // satisfied by the inventory. Mutate state.buildings + decrement inv.
    const newlyPlaceable = chain.filter((id) =>
      !placed.includes(id) && canAfford(state.inventory, BUILDING_DEFS[id].placementCost)
    );
    for (const id of newlyPlaceable) {
      payCost(state, BUILDING_DEFS[id].placementCost);
      state.buildings = [...state.buildings, freshPlacement(id)];
      placed.push(id);
      if (id === target) {
        return { reached: true, elapsedMs, placedBuildings: placed, blockedBy: null, blockingResource: null };
      }
    }
    // Advance economy.
    state = advanceIsland(state, elapsedMs + stepMs);
    elapsedMs += stepMs;
  }
  // Deadline hit; report what blocked us.
  const blockedBy = chain.find((id) => !placed.includes(id)) ?? null;
  return { reached: false, elapsedMs, placedBuildings: placed, blockedBy,
           blockingResource: blockedBy ? limitingResource(state.inventory, BUILDING_DEFS[blockedBy].placementCost) : null };
}

Helpers:

  • closurePredecessors(target): BFS over RECIPES. For each resource in BUILDING_DEFS[target].placementCost, find the recipe that produces it; for each input of that recipe, recurse. Terminate at resources present in the starter inventory.
  • selectBuildingChain(required): greedy — for each required resource, pick the lowest-tier building that produces it. Returns building ids in tier-ascending order (so T0 placed before T1, etc.).
  • canAfford(inv, cost) / payCost(state, cost): trivial inventory checks against a Partial<Record<ResourceId, number>>.
  • limitingResource(inv, cost): returns the first ResourceId where inv[r] < cost[r], for debugging.

Implementation notes

Useful context that didn't rise to the level of a spec requirement.

  • Phase 0 prerequisite ordering matters. SI-Phase 1 (resource catalog) → SI-Phase 2 (recipes) → SI-Phase 3 (mass-balance auditor) → SI-Phase 4 (this plan). If Phase 0 reveals a prerequisite hasn't landed, escalate and pause.
  • Phases 5-10 can run in parallel. Each chunk touches a disjoint subset of building-defs.ts entries; if dispatched to separate kimi sessions, they can land in any order. The plan lists them sequentially for readability; nothing in the architecture requires sequential execution.
  • Phase 2 (30 representative BOMs) must land first. It establishes the citation format and ResourceId conventions (fire_brick → clay). Phases 5-10 follow its idiom.
  • The 4 NEW defs in Phase 1 don't need Phase 2 BOM revisions. They were created with §4.3 BOMs inline. The §2.1 table lists them as NEW (Phase 1) for cross-reference, not because they need touching twice.
  • Hot-path: aggregateStorageCaps is called on every island load + every placement event. The Phase 4 refactor adds one switch statement to the per-resource loop. Negligible perf impact (the loop already runs over ALL_RESOURCES).
  • Storage building placementCost ≠ storage.capacity. Phase 3 only edits capacity; Phase 10 handles the storage buildings' BOM (placementCost). The two values change independently.
  • Per-resource baseCap is sparse intentionally. Only override what the category default gets wrong. Don't list every resource — that defeats the category default's compression value. The 4 overrides in §4.1 (plus uranium kits) are the canonical set.
  • BuildingDefId union ordering. Convention in the file appears to be alphabetical. The 4 new ids slot in at bulk_concrete_plant (before capacitor_bank), newcomen_engine (after mine-family), water_wheel (near end), windmill_t0 (right after water_wheel).
  • The reachability test runs in < 5 seconds. simulateOptimalPath uses 30 s simulation steps over a 45 min budget = 90 iterations max. Each iteration is a single advanceIsland call + a placement check. Total wall-clock well inside vitest's 5 s default per-test budget.
  • Phase 13's "live-save replay" requires /tmp/robot-islands-save.json. If the dev machine doesn't have a recent live save, dump one from the dev server (browser console: JSON.stringify(localStorage.getItem('robot-islands-save'))) and write to disk before starting Phase 0.
